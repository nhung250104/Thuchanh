Kiểm tra ngày trong tuần

Nhập số (1 → 7) để kiểm tra ngày

function checkDaysInMonth() { let month = parseInt(prompt("Nhập tháng (1-12):")); if (isNaN(month) || month < 1 || month > 12) { document.write("Tháng không hợp lệ!"); } else if (month === 2) { document.write("Tháng 2 có 28 hoặc 29 ngày (năm nhuận)."); } else if ([4, 6, 9, 11].includes(month)) { document.write("Tháng " + month + " có 30 ngày."); } else { document.write("Tháng " + month + " có 31 ngày."); } }
